        // Neural Animation Background
        const canvas = document.getElementById('neuralCanvas');
        const ctx = canvas.getContext('2d');
        let particles = [];

        function resize() {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        }

        class Particle {
            constructor() {
                this.reset();
            }
            reset() {
                this.x = Math.random() * canvas.width;
                this.y = Math.random() * canvas.height;
                this.vx = (Math.random() - 0.5) * 0.5;
                this.vy = (Math.random() - 0.5) * 0.5;
                this.size = Math.random() * 2;
            }
            update() {
                this.x += this.vx;
                this.y += this.vy;
                if (this.x < 0 || this.x > canvas.width || this.y < 0 || this.y > canvas.height) this.reset();
            }
        }

        function init() {
            resize();
            particles = Array.from({ length: 100 }, () => new Particle());
            animate();
        }

        function animate() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            ctx.strokeStyle = 'rgba(0, 219, 233, 0.1)';
            ctx.fillStyle = 'rgba(0, 219, 233, 0.4)';
            
            particles.forEach((p, i) => {
                p.update();
                ctx.beginPath();
                ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2);
                ctx.fill();

                for (let j = i + 1; j < particles.length; j++) {
                    const p2 = particles[j];
                    const dist = Math.hypot(p.x - p2.x, p.y - p2.y);
                    if (dist < 150) {
                        ctx.lineWidth = 1 - dist / 150;
                        ctx.beginPath();

        // Neural Animation Background
        const canvas = document.getElementById('neuralCanvas');
        const ctx = canvas.getContext('2d');
        let particles = [];

        function resize() {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        }

        class Particle {
            constructor() {
                this.reset();
            }
            reset() {
                this.x = Math.random() * canvas.width;
                this.y = Math.random() * canvas.height;
                this.vx = (Math.random() - 0.5) * 0.4;
                this.vy = (Math.random() - 0.5) * 0.4;
                this.size = Math.random() * 1.5;
            }
            update() {
                this.x += this.vx;
                this.y += this.vy;
                if (this.x < 0 || this.x > canvas.width || this.y < 0 || this.y > canvas.height) this.reset();
            }
        }

        function init() {
            resize();
            particles = Array.from({ length: 80 }, () => new Particle());
            animate();
        }

        function animate() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            ctx.strokeStyle = 'rgba(0, 219, 233, 0.08)';
            ctx.fillStyle = 'rgba(0, 219, 233, 0.3)';
            
            particles.forEach((p, i) => {
                p.update();
                ctx.beginPath();
                ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2);
                ctx.fill();

                for (let j = i + 1; j < particles.length; j++) {
                    const p2 = particles[j];
                    const dist = Math.hypot(p.x - p2.x, p.y - p2.y);
                    if (dist < 180) {
                        ctx.lineWidth = (1 - dist / 180) * 0.5;
                        ctx.beginPath();
                        ctx.moveTo(p.x, p.y);
                        ctx.lineTo(p2.x, p2.y);
                        ctx.stroke();
                    }
                }
            });
            requestAnimationFrame(animate);
        }

        // Active Line Scroller
        let linePos = 0;
        const activeLine = document.getElementById('activeLine');
        function updateLine() {
            linePos = (linePos + 0.3) % 100;
            activeLine.style.top = `${linePos}%`;
            requestAnimationFrame(updateLine);
        }

        window.addEventListener('resize', resize);
        init();
        updateLine();
    </script>

            // Activate timeline icons based on proximity
            const steps = document.querySelectorAll('.workflow-step');
            const icons = document.querySelectorAll('.lg\\:col-span-1 .z-20');
            
            steps.forEach((step, index) => {
                const rect = step.getBoundingClientRect();
                if (rect.top < window.innerHeight / 2 && rect.bottom > window.innerHeight / 2) {
                    icons[index]?.classList.add('border-primary-fixed-dim');
                    icons[index]?.querySelector('span').classList.add('text-primary-fixed-dim');
                } else {
                    icons[index]?.classList.remove('border-primary-fixed-dim');
                    icons[index]?.querySelector('span').classList.remove('text-primary-fixed-dim');
                }
            });
        });
